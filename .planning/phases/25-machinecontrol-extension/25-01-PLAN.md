```python
    def _write_double_word(self, register: int, value: int) -> bool:
        """Write a 32-bit value to two consecutive registers (FC16, low-word first).

        Mitsubishi low-word-first convention: register N gets low 16 bits,
        register N+1 gets high 16 bits. Uses FC16 (write_registers) for
        atomic two-register write.

        Args:
            register: Starting register address (low word destination)
            value: 32-bit integer value to write

        Returns:
            True if write succeeded
        """
        try:
            if not self._ensure_connected():
                return False

            low_word = value & 0xFFFF
            high_word = (value >> 16) & 0xFFFF

            result = self.client.write_registers(
                address=register, values=[low_word, high_word]
            )
            if result.isError():
                logger.error(
                    f"Double word write error (register {register}): {result}"
                )
                return False

            logger.debug(
                f"Double word written: register {register} = 0x{value:08X} "
                f"(low=0x{low_word:04X}, high=0x{high_word:04X})"
            )
            return True
        except Exception as e:
            logger.error(
                f"Double word write exception (register {register}): {e}"
            )
            return False
```

**Public method -- write_target_adet (PLC-01):**

```python
    def write_target_adet(self, p: int, x: int) -> bool:
        """Write target cut count to register 2050 (D2050).

        Computes P * X and writes the product as a single Word via FC6.

        Args:
            p: Number of packages (hedef paket adedi)
            x: Pieces per package (paketteki adet)

        Returns:
            True if write succeeded
        """
        try:
            value = int(p) * int(x)
            logger.debug(
                f"Writing target adet: {p} * {x} = {value} -> "
                f"register {self.TARGET_ADET_REGISTER}"
            )
            success = self._write_register(self.TARGET_ADET_REGISTER, value)
            if success:
                logger.info(f"Target adet set to {value} (P={p}, X={x})")
            return success
        except Exception as e:
            logger.error(f"Target adet write error: {e}")
            return False
```

**Public method -- write_target_uzunluk (PLC-02):**
```python
    def write_target_uzunluk(self, l_mm: float) -> bool:
        """Write target length to registers 2064-2065 (D2064/D2065) as Double Word.

        Value is scaled: l_mm * 10, then rounded to nearest integer to avoid
        IEEE 754 float imprecision. Uses FC16 write_registers with Mitsubishi
        low-word-first convention (D2064=low_word, D2065=high_word).

        Args:
            l_mm: Target length in mm (e.g., 1000.0)

        Returns:
            True if write succeeded
        """
        try:
            value = int(round(l_mm * 10))
            logger.debug(
                f"Writing target uzunluk: {l_mm}mm -> {value} (x10) -> "
                f"register {self.TARGET_LENGTH_REGISTER}"
            )
            success = self._write_double_word(self.TARGET_LENGTH_REGISTER, value)
            if success:
                logger.info(
                    f"Target uzunluk set to {l_mm}mm (modbus: {value})"
                )
            return success
        except Exception as e:
            logger.error(f"Target uzunluk write error: {e}")
            return False
```
